/**
 * 🧪 ContextClaw Eval — Build scenarios from real gateway sessions

 * Reads actual OpenClaw session transcripts (JSONL), extracts
 * multi-turn conversations, applies ContextClaw pruning, and
 * outputs paired full/pruned prompts for judge evaluation.

import { readFileSync, writeFileSync, mkdirSync, readdirSync } from 'node:fs';
import { join } from 'node:path';
import { homedir } from 'node:os';
import { encoding_for_model } from 'tiktoken';

const enc = encoding_for_model('gpt-4');
const SESSION_DIR = join(homedir(), '.openclaw', 'agents', 'main', 'sessions');
const RESULTS_DIR = join(homedir(), '.openclaw', 'workspace', 'contextclaw', 'eval', 'results');

// ---------------------------------------------------------------------------
// Scoring (replicated from plugin for standalone eval)
// ---------------------------------------------------------------------------

const STOP = new Set('the a an is are was were be been being have has had do does did will would could should may might shall can need to of in for on with at by from and but or not so if when what which who this that i me my we you your he she it they them also then here there up out yes no ok now get got make let go think know see want use'.split(' '));

function kw(text) {
  if (!text || typeof text !== 'string') return new Set();
  return new Set(text.toLowerCase().replace(/[^a-z0-9\s]/g, ' ').split(/\s+/).filter(w => w.length > 2 && !STOP.has(w)));
}

function tokens(text) {
  if (!text) return 0;
  try { return Math.ceil(enc.encode(typeof text === 'string' ? text : JSON.stringify(text)).length * 1.15); }
  catch (_) { return Math.ceil(String(text).length / 3.5); }
}

function msgTokens(msg) {
  if (!msg) return 0;
  return tokens(typeof msg.content === 'string' ? msg.content : JSON.stringify(msg.content));
}

const SAFETY_WORDS = ['never ', 'always ', "don't ", 'do not ', 'must ', 'required', 'approval', 'credential', 'password', 'constraint', 'blocker'];

function score(msg, idx, total, topicKw) {
  if (msg.role === 'system') return 999;
  let s = 0;
  s += (idx / Math.max(total - 1, 1)) * 0.4;
  s += msg.role === 'user' ? 0.25 : msg.role === 'assistant' ? 0.15 : 0.05;

  const text = typeof msg.content === 'string' ? msg.content : JSON.stringify(msg.content || '');
  const words = kw(text);
  let overlap = 0;
  for (const w of words) { if (topicKw.has(w)) overlap++; }
  s += topicKw.size > 0 ? Math.min(overlap / topicKw.size, 1) * 0.35 : 0;

  const lower = text.toLowerCase();
  if (SAFETY_WORDS.some(p => lower.includes(p))) s += 0.5;

  const t = msgTokens(msg);
  if (t > 2000) s -= 0.15;
  if (t > 5000) s -= 0.25;
  if (t > 10000) s -= 0.35;

  return s;
}

function prune(messages, targetRatio = 0.6) {
  if (messages.length < 6) return { kept: messages, evicted: [], keptTokens: 0, fullTokens: 0 };

  const lastUser = [...messages].reverse().find(m => m.role === 'user');
  const topicKw = lastUser ? kw(typeof lastUser.content === 'string' ? lastUser.content : JSON.stringify(lastUser.content)) : new Set();

  const scored = messages.map((msg, i) => ({
    msg, idx: i,
    tokens: msgTokens(msg),
    score: score(msg, i, messages.length, topicKw),
  }));

  const fullTokens = scored.reduce((s, m) => s + m.tokens, 0);
  const target = fullTokens * targetRatio;
  const protectedStart = Math.max(0, messages.length - 6);

  const candidates = [...scored]
    .filter(s => s.score < 999 && s.idx < protectedStart)
    .sort((a, b) => a.score - b.score);

  const evictedIdx = new Set();
  let current = fullTokens;
  for (const item of candidates) {
    if (current <= target) break;
    evictedIdx.add(item.idx);
    current -= item.tokens;
  }

  return {
    kept: scored.filter(s => !evictedIdx.has(s.idx)).map(s => s.msg),
    evicted: scored.filter(s => evictedIdx.has(s.idx)).map(s => s.msg),
    keptTokens: current,
    fullTokens,
    evictedCount: evictedIdx.size,
  };
}

// Parse session JSONL
// ---------------------------------------------------------------------------

function parseSession(filepath) {
  const lines = readFileSync(filepath, 'utf-8').trim().split('\n');
  const messages = [];
  for (const line of lines) {
    try {
      const entry = JSON.parse(line);
      if (entry.type !== 'message') continue;
      const msg = entry.message || entry;
      const role = msg.role;
      if (!role || !['user', 'assistant', 'system', 'tool'].includes(role)) continue;

      // Extract text from content (can be string, array of blocks, or object)
      let text = '';
      const c = msg.content;
      if (typeof c === 'string') {
        text = c;
      } else if (Array.isArray(c)) {
        text = c.map(b => {
          if (typeof b === 'string') return b;
          if (b.type === 'text') return b.text || '';
          return '';
        }).join('\n');
      } else if (c && typeof c === 'object') {
        text = JSON.stringify(c);
      }

      if (text.length > 20) {
        messages.push({ role, content: text.slice(0, 5000) });
      }
    } catch (_) {}
  }
  return messages;
}

// Build scenarios
// ---------------------------------------------------------------------------

function buildScenarios(maxScenarios = 30) {
  const files = readdirSync(SESSION_DIR)
    .filter(f => f.endsWith('.jsonl') && !f.includes('deleted'))
    .map(f => ({ name: f, path: join(SESSION_DIR, f) }));

  console.log(`[Eval] Found ${files.length} session files`);

  const scenarios = [];

  for (const { name, path } of files) {
    if (scenarios.length >= maxScenarios) break;

    const messages = parseSession(path);
    if (messages.length < 10) continue; // need substantial conversations

    // Sample from multiple points in the conversation
    const samplePoints = [
      Math.floor(messages.length * 0.25),
      Math.floor(messages.length * 0.5),
      Math.floor(messages.length * 0.75),
      messages.length - 1,
    ];

    for (const point of samplePoints) {
      if (scenarios.length >= maxScenarios) break;

      // Find a user message near this point
      let qIdx = point;
      while (qIdx >= 0 && messages[qIdx].role !== 'user') qIdx--;
      if (qIdx < 5) continue;

      const question = messages[qIdx];
      const context = messages.slice(0, qIdx);
      if (context.length < 5) continue;

      const { kept, evicted, keptTokens, fullTokens, evictedCount } = prune(context);

      // Only include if meaningful eviction happened
      if (evictedCount < 3) continue;

      scenarios.push({
        id: `s${scenarios.length + 1}`,
        source: name.slice(0, 8),
        questionIdx: qIdx,
        question: (typeof question.content === 'string' ? question.content : '').slice(0, 300),
        contextMsgs: context.length,
        prunedMsgs: kept.length,
        evictedMsgs: evictedCount,
        fullTokens,
        prunedTokens: keptTokens,
        reduction: Math.round((1 - keptTokens / fullTokens) * 100),
        fullContext: context.slice(-20), // last 20 for storage
        prunedContext: kept.slice(-20),
      });
    }
  }

  return scenarios;
}

const scenarios = buildScenarios(30);

for (const s of scenarios) {
  console.log(`${s.id.padEnd(4)}| ${s.source.padEnd(9)}| ${String(s.contextMsgs).padEnd(9)}| ${String(s.prunedMsgs).padEnd(7)}| ${String(s.evictedMsgs).padEnd(8)}| ${String(s.fullTokens).padEnd(12)}| ${String(s.prunedTokens).padEnd(9)}| ${s.reduction}%`);
}

const totalFull = scenarios.reduce((s, x) => s + x.fullTokens, 0);
const totalPruned = scenarios.reduce((s, x) => s + x.prunedTokens, 0);
console.log(`\nAggregate: ${totalFull.toLocaleString()} → ${totalPruned.toLocaleString()} tokens (${Math.round((1 - totalPruned/totalFull) * 100)}% reduction)`);

const outFile = join(RESULTS_DIR, `eval-${new Date().toISOString().replace(/[:.]/g, '-')}.json`);
writeFileSync(outFile, JSON.stringify({ scenarios, aggregate: { totalFull, totalPruned, reduction: Math.round((1 - totalPruned/totalFull) * 100), scenarioCount: scenarios.length } }, null, 2));
console.log(`\n[Eval] Results: ${outFile}`);
